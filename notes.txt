# neu-comfy-ui-podman
# Build image with ContainerFile:
# podman build -t comfyui-neu:latest .
# podman build -t comfyui-neuv2:latest .

# To start container normally (access via http)
podman run -it --rm \
  --name comfyui-max \
  --device nvidia.com/gpu=all \
  --security-opt label=disable \
  --userns=keep-id \
  -p 8188:8188 \
  -e COMFYUI_USE_XFORMERS=1 \
  -e TORCH_BACKEND_ENABLE_CUDA_FLASH_ATTENTION=1 \
  -e PYTORCH_CUDA_ALLOC_CONF=max_split_size_mb:256 \
  -v /home/arrel/.cache/huggingface:/root/.cache/huggingface \
  -v ~/comfyui/models:/workspace/ComfyUI/models:Z \
  -v ~/comfyui/output:/workspace/ComfyUI/output:Z \
  -v ~/comfyui/custom_nodes:/workspace/ComfyUI/custom_nodes:Z \
  -v ~/comfyui/settings:/workspace/ComfyUI/user:Z \
  -v ~/comfyui/temp:/workspace/ComfyUI/temp:Z \
  localhost/comfyui-neu:latest

# If you see an error like:
# Error: crun: cannot stat `/usr/lib/x86_64-linux-gnu/libnvidia-egl-wayland.so.1.1.21`: No such file or directory: OCI runtime attempted to invoke a command that was not found
# 
# It probably means that the nvidia updated since the previous run and you need to 
# update the nvidia container toolkit: "sudo nvidia-ctk cdi generate --output=/etc/cdi/nvidia.yaml"
